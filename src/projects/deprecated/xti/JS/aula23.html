Curso de JavaScript - Aula 23 - SPLIT / JOIN

SPLIT

.split() = Função que permite a divisão de strings em um Array, recebe como param o valor de quebra.

Exemplo

let [a,b,c,d] = "um texto de Exemplo".split(" ");
console.log(a);
console.log(b);
console.log(c);
console.log(d);

retorno:

um texto de exemplo

JOIN

.join() = Função que permite a concatenação de strings advinda de um Array, recebe como param o valor de quebra.

Exemplo

let a = "outro"
let a = "texto"
let a = "de"
let a = "exemplo"
let frase = [a,b,c,d].join(" ");

retorno:

outro texto de exemplo
